20 /** Orders the names by their score. If scores are the same, order by their
21  * last name followed by first name
22  */
23 struct s_record_order
24 {
25  /** Comparison operator for s_record. The value returned indicates whether
26  * the element passed as first argument is considered to go before the
27  * second in the specific strict weak ordering it defines.
28  * @param[in] r1 First record to be compared
29  * @param[in] r2 Second record to be compared
30  * @return TRUE for r1 to be ordered before r2, FALSE for r2 to be ordered
31  * before r1.
32  */
33  inline bool operator() (const s_record &r1, const s_record &r2)
34  {
35  if (r1.score == r2.score)
36  {
37  /* Scores are equal, order by last then first name */
38  int c = strcmp(r1.llast.data(), r2.llast.data());
39  if (c == 0)
40  {
41  /* Last names are equal, order by first name */
42  return(strcmp(r1.lfirst.data(), r2.lfirst.data()) < 0);
43  }
44  return(c < 0);
45  }
46  return(r1.score > r2.score);
47  }
48 } o_record_order;

70 bool validate_arg(const int argc, char **argv)
71 {
72  size_t l; /* Length of source filename */

75  char fext[PATH_MAX]; /* File name post dot */
76 
77  /* Initialist the destination path global and local variables */
78  g_ofname[0] = '\0';
79  fname[0] = '\0';
80  fext[0] = '\0';
81  /* Make sure we only have one command line parameter */
82  if (argc > (MAX_ARGS+1))
83  {
84  print_error("This application only accepts one command line parameter");
85  return(false);
86  }
87  if (argc < (MAX_ARGS+1))
88  {
89  print_error("This application requires one command line parameter");
90  return(false);
91  }
92  /* Make sure the filename parameter does not exceed system limits */
93  l = strlen(SRC);
94  if (l == 0)
95  {
96  print_error("Source file name is required");
97  return(false);
98  }
99  if (l >= PATH_MAX-1)
100  {
101  print_error("Source file name length exceeds system limits");
102  return(false);
103  }
104  if (l + strlen(FOUT_EXT) >= PATH_MAX-1)
105  {
106  print_error("Destination file name length exceeds system limits");
107  return(false);
108  }
109  /* Make sure the file exists and is readable */
110  if (access(SRC, R_OK)!=0)
111  {
112  SYSERR("File access error");
113  return(false);
114  }
115  /* The path might have '/' or '\' so break out the path and filename
116  * for separate processing */
117  std::string src(SRC);
118  ld = src.find_last_of("/\\");
119  if (ld == std::string::npos)
120  ld = 0;
121  else
122  {
123  ++ld;
124  strncpy(g_ofname, SRC, ld);
125  g_ofname[ld]='\0';
126  }
127  g_ofshort = ld;
128  /* Count any leading dots in filename and append dots to g_ofname */
129  for (; (ld < l) && (SRC[ld]=='.'); ++ld) strcat(g_ofname,".");
130  /* Break the filename into its components */
131  sscanf(SRC+ld, "%[^.].%s", fname, fext);
132  /* Build the destination file name */
133  strcat(g_ofname, fname);
134  strcat(g_ofname, FOUT_EXT);
135  if (strlen(fext))

139  }
140  return(true);

152 e_colcode CSimpleCSV::read_col(std::ifstream &file, std::string &v)
153 {
154  char c;
155  v="";
156  while (1)
157  {
158  switch (c=file.get())
159  {
160  case EOF:
161  return(colcode_EOF);
162  case '\r':
163  if (file.peek()=='\n') file.get();
164  return(colcode_EOL);
165  case '\n':
166  if (file.peek()=='\r') file.get();
167  return(colcode_EOL);
168  case ',':
169  return(colcode_OK);
170  default:
171  if (file.good())
172  v.push_back(c);
173  else
174  return(colcode_FAIL);
175  }
176  }
177  /* Never happens */
178  return(colcode_OK);

181 bool CSimpleCSV::read_row(std::ifstream &file)
182 {
183  unsigned char col; /* CSV column counter */
184  std::string v; /* Raw column value from file */
185  /* Leading rows must have a comma */
186  for (col=0; col<FCOL_MAX-1; ++col)
187  {
188  if (read_col(file, v)==colcode_OK)
189  m_value[col] = trim(v);
190  else
191  return(false);
192  }
193  /* Last row must not have a trailing comma or additional columns */
194  switch (read_col(file, v))
195  {
196  case colcode_EOL:
197  case colcode_EOF:
198  m_value[col] = trim(v);
199  break;
200  case colcode_OK:
201  /* Drain remaining characters on row */
202  while (read_col(file, v)==colcode_OK);
203  return(false);
204  default:
205  return(false);
206  }
207  return(true);
208 }
